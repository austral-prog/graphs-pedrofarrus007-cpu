Line Plot¶

Este gráfico muestra los cambios de temperatura a lo largo de una semana

In [ ]:
days = ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"]
temperature = [20, 22, 19, 24, 26, 25, 23]

plt.figure(figsize=(8,5))

plt.plot(
    days,
    temperature,
    color='blue',
    marker='o',
    linestyle='--',
    label='Temperature'
)

plt.title("Weekly Temperature Trends")
plt.xlabel("Days")
plt.ylabel("Temperature (°C)")
plt.legend()

plt.savefig('imgs/line_plot.png', dpi=300, bbox_inches='tight')

plt.show()

activities = ["Sleeping", "Studying", "Gaming", "Exercise", "Eating"]
hours_spent = [8, 6, 4, 2, 4]

plt.figure(figsize=(8,8))

plt.pie(
    hours_spent,
    labels=activities,
    autopct='%1.1f%%',
    startangle=90
)

plt.title("Daily Activities Distribution")

plt.savefig('imgs/custom_plot.png', dpi=300, bbox_inches='tight')
